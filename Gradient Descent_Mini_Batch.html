# step for linear regression, following a squared error rule. See the docstring
# for parameters and returned variables.
def MSEStep(X, y, W, b, learn_rate = 0.005):
    """
    This function implements the gradient descent step for squared error as a
    performance metric.
    
    Parameters
    X : array of predictor features
    y : array of outcome values
    W : predictor feature coefficients
    b : regression function intercept
    learn_rate : learning rate

    Returns
    W_new : predictor feature coefficients following gradient descent step
    b_new : intercept following gradient descent step
    """
    
    # Fill in code
    
    # compute errors                             # matmul(W, X) 실행 시 에러 발생. c.f.) Zero 행렬을 앞에 위치시키면 에러!
    y_pred = np.matmul(X, W) + b                 # [y^ = WX + b] 초기값, X: e.g. array([-2.40724]) / W = array([0.]) / b = 0
    error = y - y_pred                           # Error[w1으로 편미분] = -(y - y^)x 에서  x를 제외한 -(y - y^)부분
    
    # compute steps
    W_new = W + learn_rate * np.matmul(error, X) # 기존 기울기에 새로운 Error 편분값 [-(y - y^)x]*alpha 추가
    b_new = b + learn_rate * error.sum()         # Error 편미분값 배열을 float 형태로 합계
    return W_new, b_new
   

# The parts of the script below will be run when you press the "Test Run"
# button. The gradient descent step will be performed multiple times on
# the provided dataset, and the returned list of regression coefficients
# will be plotted.
def miniBatchGD(X, y, batch_size = 20, learn_rate = 0.005, num_iter = 25):
    """
    This function performs mini-batch gradient descent on a given dataset.

    Parameters
    X : array of predictor features
    y : array of outcome values
    batch_size : how many data points will be sampled for each iteration
    learn_rate : learning rate
    num_iter : number of batches used

    Returns
    regression_coef : array of slopes and intercepts generated by gradient
      descent procedure
    """
    n_points = X.shape[0]
    W = np.zeros(X.shape[1]) # coefficients                   # W 초기값: array([0.])
    b = 0 # intercept                                         # y^ = WX + b 에 대한 초기값 
    
    # run iterations
    regression_coef = [np.hstack((W,b))]                      # [기울기, 절대값] 으로 행렬 병합 
    for _ in range(num_iter):                                 # Mini batch 25개 반복
        batch = np.random.choice(range(n_points), batch_size) # 100개 가운데 무작위 batch_size(20개)만큼 착출
        X_batch = X[batch,:]                                  # x[100행 1열]가운데 nth행        초기값 e.g. array([-2.40724])
        y_batch = y[batch]                                    # y[1행 100열]리스트 번호로 축출   초기값 e.g. -0.00156
        W, b = MSEStep(X_batch, y_batch, W, b, learn_rate)    
        regression_coef.append(np.hstack((W,b)))              # [기울기, 절대값] 각각 gathering
    
    return regression_coef         # 초기값 array([0., 0.]),array([0.12341158, 0.2049234 ]),array([0.07713629, 0.37119546]),

    # perform gradient descent
    data = np.loadtxt('data.csv', delimiter = ',')
    X = data[:,:-1]                                           # '-1'은 가장 마지막에 위치한 칼럼이다. 
    y = data[:,-1]                                            # '열' 형태로 나열. 즉, x[100행 1열] * y[1행 100열]
    regression_coef = miniBatchGD(X, y)                       # Gradient Descent 가 적용된 보정값 개선치 배열

    counter = len(regression_coef)                            # 총 26개

        counter -= 1
        color = [1 - 0.92 ** counter for _ in range(3)]
        plt.plot([X_min, X_max],[X_min * W + b, X_max * W + b], color = color) # plt.plot([x축값1, x축값2], [y축값1, y축값2])

    plt.show()
